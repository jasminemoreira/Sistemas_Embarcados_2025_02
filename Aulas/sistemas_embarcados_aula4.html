int calcularRAMlivre() {
  // Truque para calcular RAM livre
  extern int __heap_start, *__brkval;
  int v;
  return (int) &v - (__brkval == 0 ? (int) &__heap_start : (int) __brkval);
}

void mostrarMemoria(String momento) {
  int ram_livre = calcularRAMlivre();
  int ram_usada = 8192 - ram_livre;
  float percentual_uso = (ram_usada / 8192.0) * 100;
  
  Serial.print("üìä ");
  Serial.print(momento);
  Serial.println(":");
  Serial.print("   RAM Livre: ");
  Serial.print(ram_livre);
  Serial.println(" bytes");
  Serial.print("   RAM Usada: ");
  Serial.print(ram_usada);
  Serial.print(" bytes (");
  Serial.print(percentual_uso, 1);
  Serial.println("%)");
  Serial.println();
}
